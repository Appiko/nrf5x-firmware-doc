180  LOGS[log_config->log_id].no_pages = log_config->no_of_pages;

182  {

184  - page_no*NVM_LOGGER_PAGE_OFFSETS;
185  }
186  LOGS[log_config->log_id].last_entry_no = (BYTES_PER_PAGE/(LOGS[log_config->log_id].entry_size * WORD_SIZE));
